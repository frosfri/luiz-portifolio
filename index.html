// components/Cursor.jsx // Cursor personalizado que segue o mouse import { useEffect, useRef } from 'react' export default function Cursor() { const curRef = useRef(null) useEffect(() => { const el = curRef.current if (!el) return // Atualiza posição do cursor const onMove = e => { el.style.left = (e.clientX - 7) + 'px' el.style.top = (e.clientY - 7) + 'px' } // Efeito de clique const onDown = () => el.classList.add('clicked') const onUp = () => el.classList.remove('clicked') document.addEventListener('mousemove', onMove) document.addEventListener('mousedown', onDown) document.addEventListener('mouseup', onUp) // Limpeza: remove os eventos quando o componente sair da tela return () => { document.removeEventListener('mousemove', onMove) document.removeEventListener('mousedown', onDown) document.removeEventListener('mouseup', onUp) } }, []) return
}
